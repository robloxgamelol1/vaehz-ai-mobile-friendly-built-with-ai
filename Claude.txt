-- ── Number decoder ───────────────────────────────────────────────
local SFX = {K=1e3,M=1e6,B=1e9,T=1e12,Qd=1e15,Qn=1e18,
             Sx=1e21,Sxd=1e21,Sp=1e24,Oc=1e27,No=1e30,Dc=1e33}
local function dec(s)
    if not s then return nil end
    local clean = tostring(s):gsub("[\226\128\128-\226\128\143]","")
    local n, sfx = clean:match("%$([%d%,%.]+)(%a*)")
    if not n then return nil end
    n = tonumber(n:gsub(",",""))
    if sfx == "" then return n end
    local m = SFX[sfx] or SFX[sfx:sub(1,1):upper()..sfx:sub(2):lower()]
    return m and n*m or n
end

    local c = Instance.new("UICorner")

local function pad(f, t, b, l, r)
    local p = Instance.new("UIPadding")
    p.PaddingTop    = UDim.new(0, t or 0); p.PaddingBottom = UDim.new(0, b or 0)
    p.PaddingLeft   = UDim.new(0, l or 0); p.PaddingRight  = UDim.new(0, r or 0)

-- Toggle card  (icon | label | iOS-style switch)
local function toggle(parent, ico, label, def, cb, order)
    local card = Instance.new("Frame"); card.LayoutOrder = order
    card.Size = UDim2.new(1,0,0,58); card.BackgroundColor3 = C.card
    card.BorderSizePixel = 0; card.ZIndex = 12; card.Parent = parent; crn(card,12)

    -- icon
    local iLbl = Instance.new("TextLabel"); iLbl.Text = ico
    iLbl.Size = UDim2.new(0,40,1,0); iLbl.Position = UDim2.new(0,8,0,0)
    iLbl.BackgroundTransparency = 1; iLbl.TextSize = 20; iLbl.ZIndex = 13; iLbl.Parent = card

    -- label
    local tLbl = Instance.new("TextLabel"); tLbl.Text = label
    tLbl.Size = UDim2.new(1,-112,1,0); tLbl.Position = UDim2.new(0,52,0,0)
    tLbl.BackgroundTransparency = 1; tLbl.TextColor3 = C.txt; tLbl.TextSize = 14
    tLbl.Font = Enum.Font.Gotham; tLbl.TextXAlignment = Enum.TextXAlignment.Left
    tLbl.ZIndex = 13; tLbl.Parent = card

    -- switch track
    local track = Instance.new("Frame"); track.Size = UDim2.new(0,50,0,28)
    track.Position = UDim2.new(1,-60,0.5,-14)
    track.BackgroundColor3 = def and C.grn or C.off
    track.BorderSizePixel = 0; track.ZIndex = 13; track.Parent = card; crn(track,14)

    -- switch knob
    local knob = Instance.new("Frame"); knob.Size = UDim2.new(0,22,0,22)
    knob.Position = def and UDim2.new(1,-24,0.5,-11) or UDim2.new(0,3,0.5,-11)
    knob.BackgroundColor3 = C.wht; knob.BorderSizePixel = 0
    knob.ZIndex = 14; knob.Parent = track; crn(knob,11)

    -- invisible full-card tap area
    local hit = Instance.new("TextButton"); hit.Size = UDim2.new(1,0,1,0)
    hit.BackgroundTransparency = 1; hit.Text = ""; hit.ZIndex = 16; hit.Parent = card

    local state = def
    hit.MouseButton1Click:Connect(function()
        state = not state; cb(state)
        TweenService:Create(track, tw, {BackgroundColor3 = state and C.grn or C.off}):Play()
        TweenService:Create(knob,  tw, {
            Position = state and UDim2.new(1,-24,0.5,-11) or UDim2.new(0,3,0.5,-11)
        }):Play()
    end)
    return card
end

-- ================================================================
--  DRAG: Floating Button  (drag to move, tap to toggle panel)
-- ================================================================

do
    local dn,di,sp,sm,mv = false,nil,nil,nil,false

    FB.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.Touch
        or i.UserInputType == Enum.UserInputType.MouseButton1 then
            dn=true; mv=false; sp=FB.Position; sm=i.Position; di=i
        end
    end)

    FB.InputEnded:Connect(function(i)
        if i == di then
            dn=false
            if not mv then setPanel(not pOpen) end -- tap = toggle
            mv=false
        end
    end)

    UserInputService.InputChanged:Connect(function(i)
        if not dn then return end
        if i.UserInputType==Enum.UserInputType.Touch
        or i.UserInputType==Enum.UserInputType.MouseMovement then
            local d = i.Position - sm
            if d.Magnitude > 8 then
                mv = true
                FB.Position = UDim2.new(sp.X.Scale, sp.X.Offset+d.X,
                                        sp.Y.Scale, sp.Y.Offset+d.Y)
            end
        end
    end)
end

local farming = false
local PF = tycoon.Purchases

tycoon.Remotes.PhoneOffer.OnClientEvent:Connect(function()
    if not farming then return end
    tycoon.Remotes.PhoneOffer:FireServer("Accept")
end)

local function setFarm(on)
    farming = on; getgenv().farming = on
    StartBtn.Text             = on and "⏹  Stop Autofarm" or "▶  Start Autofarm"
    StartBtn.BackgroundColor3 = on and C.red              or C.acc
    SDot.BackgroundColor3     = on and C.grn              or C.sub
    SLbl.Text                 = on and "Farming active…"  or "Idle"
    SLbl.TextColor3           = on and C.grn              or C.sub
end

local function farmLoop()
    local streams = tycoon.Values.Income.Streams

    -- Collect thread runs independently so it doesn't block purchases
    task.spawn(function()
        while farming do
            if getgenv().farmsettings.collect then
                for _, v in pairs(streams:GetChildren()) do
                    tycoon.Remotes.WakeIncomeStream:InvokeServer(v.Name)
                end
            end
            task.wait()
        end
    end)

    while farming do
        -- Purchase
        pcall(function()
            if not getgenv().farmsettings.purchase then return end
            for _, fold in pairs(PF:GetChildren()) do
                if not fold:FindFirstChild("Buttons") then continue end
                for _, nf in pairs(fold.Buttons:GetChildren()) do
                    local list = nf:IsA("Folder") and nf:GetChildren()
                              or (nf:IsA("Model") and {nf} or {})
                    for _, btn in pairs(list) do
                        if btn:GetAttribute("Shown") and btn:GetAttribute("Enabled")
                        and not btn:GetAttribute("Purchased") then
                            local price = dec(btn.Button.Gui.Price.Text)
                            local bal   = dec(plr.leaderstats.Cash.Value)
                            if price and bal and price <= bal then
                                firetouchinterest(plr.Character.Head, btn.Button, true)
                                task.wait()
                                firetouchinterest(plr.Character.Head, btn.Button, false)
                            end
                        end
                    end
                end
            end
        end)

        -- Upgrade
        pcall(function()
            if not getgenv().farmsettings.upgrade then return end
            for _, fold in pairs(PF:GetChildren()) do
                local inner = fold:FindFirstChild(fold.Name)
                if inner and inner:GetAttribute("Enabled") then
                    inner:FindFirstChild(fold.Name).Upgrade:InvokeServer(1)
                end
            end
        end)

        -- Cash drops
        pcall(function()
            if not getgenv().farmsettings.cashdrop then return end
            for _, v in pairs(workspace.CashDrops:GetChildren()) do
                firetouchinterest(plr.Character.Head, v, true)
                task.wait()
                firetouchinterest(plr.Character.Head, v, false)
            end
        end)

        -- Fruit
        pcall(function()
            if not getgenv().farmsettings.fruit then return end
            for _, v in pairs(tycoon.Constant.Trees:GetChildren()) do
                for _, f in pairs(v:GetChildren()) do
                    if f.Name == "Fruit" and f:FindFirstChild("ClickPart") then
                        fireclickdetector(f.ClickPart.ClickDetector)
                        task.wait()
                    end
                end
            end
        end)

        task.wait(1)
    end
end

StartBtn.MouseButton1Click:Connect(function()
    if not farming then
        setFarm(true); task.spawn(farmLoop)
    else
        setFarm(false)
    end
end)
